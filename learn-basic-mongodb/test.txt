> db.dogs.insert({
  name: 'John',
  breed: 'Mutt'
})

> db.dogs.find()
> db.dogs.find({ name: 'John' })

// update data collection
db.dogs.update({
  name: 'John'
}, { breed: 'Labradoodle' }) // will overide all objects

db.dogs.update({
  name: 'John'
}, {$set: { breed: 'Labradoodle', isCute: true } }) // not overide and add new data isCute

db.dogs.remove({
  breed: 'Mutt'
})
